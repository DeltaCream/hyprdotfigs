    fn set_clipboard(content: &[u8], mime_type: &str) -> Result<(), SherlockError> {
        let mut child = Command::new("wl-copy")
            .arg("--type")
            .arg(mime_type)
            .stdin(Stdio::piped())
            .spawn()
            .map_err(|e| {
                sherlock_error!(
                    SherlockErrorType::CommandExecutionError("wl-copy".to_string()),
                    e.to_string()
                )
            })?;

        if let Some(mut stdin) = child.stdin.take() {
            stdin.write_all(content).map_err(|e| {
                sherlock_error!(
                    SherlockErrorType::FileWriteError(PathBuf::from("stdin")),
                    e.to_string()
                )
            })?;
        }

        child.wait().map_err(|e| {
            sherlock_error!(
                SherlockErrorType::CommandExecutionError("wl-copy".to_string()),
                e.to_string()
            )
        })?;

        Ok(())
    }

impl ClipboardParser {
    fn parse(
        launcher: Arc<Launcher>,
        db_path: PathBuf,
        max_entries: usize,
        show_images: bool,
    ) -> Result<Vec<AppData>, SherlockError> {
        use rusqlite::Connection;

        let mut res: Vec<AppData> = Vec::new();

        // Connect to clipvault database
        let conn = Connection::open(&db_path)
            .map_err(|e| sherlock_error!(SherlockErrorType::SqlConnectionError(), e.to_string()))?;

        // Query to get recent clipboard entries
        let query = if show_images {
            format!(
                "SELECT id, content, mime_type, timestamp
                FROM clipboard_history
                ORDER BY timestamp DESC
                LIMIT {}",
                max_entries
            )
        } else {
            format!(
                "SELECT id, content, mime_type, timestamp
                FROM clipboard_history
                WHERE mime_type LIKE 'text/%'
                ORDER BY timestamp DESC
                LIMIT {}",
                max_entries
            )
        };

        if let Ok(mut stmt) = conn.prepare(&query) {
            let entry_iter = stmt.query_map([], |row| {
                let id: i64 = row.get(0)?;
                let content: Vec<u8> = row.get(1)?;
                let mime_type: String = row.get(2)?;
                let timestamp: i64 = row.get(3)?;

                Ok((id, content, mime_type, timestamp))
            });

            if let Ok(rows) = entry_iter {
                for (index, row) in rows.flatten().enumerate() {
                    let (id, content, mime_type, timestamp) = row;

                    // Create preview for the entry
                    let (title, icon_name) = Self::create_preview(&content, &mime_type);

                    // Format timestamp
                    let time_str = Self::format_timestamp(timestamp);

                    // Create subtitle with mime type and time
                    let subtitle = format!("{} • {}", mime_type, time_str);

                    let clipboard_entry = AppData {
                        name: Some(SharedString::from(&title)),
                        icon: resolve_icon_path(&icon_name),
                        search_string: format!(
                            // "{} {} {} {}",
                            // title.to_lowercase(),
                            // mime_type.to_lowercase(),
                            // time_str.to_lowercase(),
                            // id
                            "{} {} {}",
                            title.to_lowercase(),
                            subtitle.to_lowercase(),
                            id
                        ),
                        // Store the ID in exec field for retrieval
                        exec: Some(format!("clipboard:{}", id)),
                        desktop_file: None,
                        priority: Some(
                            launcher.priority as f32
                                + (max_entries - index) as f32 / max_entries as f32,
                        ),
                        actions: Arc::new([]),
                        vars: vec![],
                        terminal: false,
                    };

                    res.push(clipboard_entry);
                }
            }
        }

        Ok(res)
    }

    fn create_preview(content: &[u8], mime_type: &str) -> (String, String) {
        if mime_type.starts_with("text/") {
            let text = String::from_utf8_lossy(content);
            let preview = text
                .lines()
                .next()
                .unwrap_or("")
                .chars()
                .take(100)
                .collect::<String>();

            let icon = if mime_type.contains("html") {
                "text-html"
            } else {
                "edit-copy"
            };

            (preview, icon.to_string())
        } else if mime_type.starts_with("image/") {
            let icon = "image-x-generic";
            (format!("📷 Image ({})", mime_type), icon.to_string())
        } else {
            let icon = "edit-copy";
            (
                format!("📎 {} ({} bytes)", mime_type, content.len()),
                icon.to_string(),
            )
        }
    }

    fn format_timestamp(timestamp: i64) -> String {
        use chrono::{DateTime, Local, Utc};

        if let Some(datetime) = DateTime::from_timestamp(timestamp, 0) {
            let local_time: DateTime<Local> = datetime.with_timezone(&Local);
            let now = Local::now();

            let duration = now.signed_duration_since(local_time);

            if duration.num_seconds() < 60 {
                "Just now".to_string()
            } else if duration.num_minutes() < 60 {
                format!("{}m ago", duration.num_minutes())
            } else if duration.num_hours() < 24 {
                format!("{}h ago", duration.num_hours())
            } else if duration.num_days() < 7 {
                format!("{}d ago", duration.num_days())
            } else {
                local_time.format("%b %d").to_string()
            }
        } else {
            "Unknown".to_string()
        }
    }
}
